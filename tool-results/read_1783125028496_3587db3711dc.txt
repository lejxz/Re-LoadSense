    37→def normalize_country(country: str | None = None) -> str:
    38→    if not country:
    39→        return "PH"
    40→    code = country.strip().upper()
    41→    if code in COUNTRY_CODES:
    42→        return code
    43→    if code in _COUNTRY_NAME_TO_CODE:
    44→        return _COUNTRY_NAME_TO_CODE[code]
    45→    return "PH"

    58→def _get_connection(country: str | None = None) -> sqlite3.Connection:
    59→    """Return the country-specific connection, creating it if needed."""
    60→    code = normalize_country(country)
    61→    if code not in _connections:
    62→        path = database_path(code)
    63→        path.parent.mkdir(parents=True, exist_ok=True)
    64→        conn = sqlite3.connect(path, check_same_thread=False)
    65→        conn.row_factory = sqlite3.Row
    66→        conn.execute("PRAGMA journal_mode=WAL;")
    67→        conn.execute("PRAGMA foreign_keys=ON;")
    68→        conn.execute("PRAGMA synchronous=NORMAL;")
    69→        conn.execute("PRAGMA cache_size=-8000;")  # 8MB cache
    70→        _connections[code] = conn
    71→    return _connections[code]

   278→def _seed_routes_if_needed(conn: sqlite3.Connection, country_code: str) -> None:
   279→    """Seed routes from GeoJSON files if the database is empty."""
   280→    row = conn.execute("SELECT COUNT(*) AS count FROM routes").fetchone()
   281→    if int(row["count"] if row else 0) > 0:
   282→        return
   283→    total = 0
   284→    geojson_path = route_geojson_path(country_code)
   285→    if not geojson_path.exists():
   286→        return
   287→    routes = _load_geojson_routes(geojson_path, country_code, max_routes=50, max_points=120)
   288→    for route in routes:
   289→        conn.execute(
   290→            """
   291→            INSERT OR IGNORE INTO routes (
   292→                route, name, polyline_json, country, region,
   293→                tag, route_type, origin_name, destination_name, distance_km, description, minimum_fare, fare_per_km
   294→            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, NULL, NULL, NULL, NULL)
   295→            """,
   296→            (
   297→                route["route"],
   298→                route["name"],
   299→                json.dumps(route["polyline"]),
   300→                country_code,
   301→                route.get("region") or route.get("city") or "",
   302→                route.get("tag") or route["route"],
   303→                route.get("route_type") or "PUV",
   304→                route.get("origin_name") or "",
   305→                route.get("destination_name") or "",
   306→            ),
   307→        )
   308→        total += 1
   309→    if total > 0:
   310→        conn.commit()

   313→def _load_geojson_routes(
   314→    path, country: str, max_routes: int = 50, max_points: int = 120
   315→) -> list[dict[str, Any]]:
   316→    """Load routes from a GeoJSON file. Limits to max_routes for performance."""
   317→    try:
   318→        payload = json.loads(path.read_text(encoding="utf-8"))
   319→    except Exception:
   320→        return []
   321→    routes: list[dict[str, Any]] = []
   322→    for index, feature in enumerate(payload.get("features", [])):
   323→        if len(routes) >= max_routes:
   324→            break
   325→        props = feature.get("properties", {})
   326→        geometry = feature.get("geometry", {})
   327→        if geometry.get("type") != "LineString":
   328→            continue
   329→        route_id = str(props.get("route_id") or props.get("route") or f"{country}-{index + 1}").strip()
   330→        name = str(props.get("route_name") or props.get("name") or route_id).strip()
   331→        # GeoJSON is [lon, lat], convert to (lat, lon)
   332→        raw_coords = geometry.get("coordinates", [])
   333→        points = [(float(c[1]), float(c[0])) for c in raw_coords if len(c) >= 2]
   334→        points = _sample_polyline(points, max_points=max_points)
   335→        if route_id and name and len(points) >= 2:
   336→            # Infer city from first coordinate
   337→            city = infer_city(
   338→                [{"latitude": points[0][0], "longitude": points[0][1]}],
   339→                name,
   340→            )
   341→            routes.append({
   342→                "route": route_id,
   343→                "name": name,
   344→                "polyline": points,
   345→                "city": city or "",
   346→                "province": props.get("province") or "",
   347→                "region": props.get("region") or "",
   348→                "tag": props.get("route_ref") or props.get("ref") or route_id,
   349→                "route_type": props.get("route_type") or "PUV",
   350→                "origin_name": props.get("origin_name") or "",
   351→                "destination_name": props.get("destination_name") or "",
   352→            })
   353→    return routes
   354→
   355→
   356→def _sample_polyline(points: list[tuple[float, float]], max_points: int) -> list[tuple[float, float]]:
   357→    if len(points) <= max_points:
   358→        return points
   359→    sampled: list[tuple[float, float]] = []
   360→    last_index = len(points) - 1
   361→    for i in range(max_points):
   362→        sampled.append(points[round((i / (max_points - 1)) * last_index)])
   363→    result: list[tuple[float, float]] = []
   364→    for point in sampled:
   365→        if not result or point != result[-1]:
   366→            result.append(point)
   367→    return result

   374→def get_regions_for_country(country: str) -> list[str]:
   375→    """Return distinct region entries for routes in a country.
   376→    Used to populate region dropdowns dynamically."""
   377→    code = normalize_country(country)
   378→    init_db(code)
   379→    conn = _get_connection(code)
   380→    rows = conn.execute(
   381→        """
   382→        SELECT DISTINCT region
   383→        FROM routes
   384→        WHERE country = ?
   385→          AND COALESCE(region, '') != ''
   386→          AND UPPER(region) <> UPPER(country)
   387→        ORDER BY region
   388→        """,
   389→        (code,),
   390→    ).fetchall()
   391→    return [row["region"] for row in rows]

   406→def country_for_route(route: str, default: str = "PH") -> str:
   407→    route_text = str(route or "")
   408→    for code in COUNTRY_CODES:
   409→        init_db(code)
   410→        conn = _get_connection(code)
   411→        row = conn.execute("SELECT country FROM routes WHERE route = ? OR tag = ? LIMIT 1", (route_text, route_text)).fetchone()
   412→        if row and row["country"]:
   413→            return normalize_country(row["country"])
   414→    return normalize_country(default)

   535→def verify_alert(alert_id: str, action: str, note: str, timestamp: str) -> Optional[OperatorAlert]:
   536→    normalized_action = action if action in {"verified", "false_alarm", "escalated"} else "verified"
   537→    acknowledged = 0 if normalized_action == "escalated" else 1
   538→    conn = None
   539→    row = None
   540→    for code in COUNTRY_CODES:
   541→        init_db(code)
   542→        candidate = _get_connection(code)
   543→        row = candidate.execute("SELECT * FROM operator_alerts WHERE id = ?", (alert_id,)).fetchone()
   544→        if row is not None:
   545→            conn = candidate
   546→            break
   547→    if row is None:
   548→        return None
   549→    conn.execute(
   550→        """
   551→        UPDATE operator_alerts

   775→def database_status(country: str | None = None) -> dict[str, Any]:
   776→    countries = [normalize_country(country)] if country else list(COUNTRY_CODES)
   777→    tables = ["telemetry_logs", "vehicle_states", "operator_alerts", "operator_feedback", "chatbot_queries", "routes", "route_points", "vehicles"]
   778→    counts = {table: 0 for table in tables}
   779→    load_rows: list[dict[str, Any]] = []
   780→    vehicle_rows: list[dict[str, Any]] = []
   781→    alert_totals: dict[str, int] = {}
   782→    recent_chats: list[dict[str, Any]] = []
   783→    country_statuses: dict[str, Any] = {}
   784→    for code in countries:
   785→        init_db(code)
   786→        conn = _get_connection(code)
   787→        country_counts = {
   788→            table: conn.execute(f"SELECT COUNT(*) AS count FROM {table}").fetchone()["count"]
   789→            for table in tables
   790→        }
   791→        for table, count in country_counts.items():
   792→            counts[table] += int(count)
   793→        country_statuses[code] = {
   794→            "path": str(database_path(code)),
   795→            "exists": database_path(code).exists(),
   796→            "tables": country_counts,
   797→        }
   798→        load_rows.extend(dict(row) | {"country": code} for row in conn.execute(
   799→            """
   800→            WITH recent_logs AS (
   801→                SELECT route, occupancy, tier
   802→                FROM telemetry_logs
   803→                ORDER BY id DESC
   804→                LIMIT 5000
   805→            )
   806→            SELECT route,
   807→                   COUNT(*) AS samples,
   808→                   ROUND(AVG(occupancy), 2) AS average_occupancy,
   809→                   SUM(CASE WHEN tier = 'blinking_red' THEN 1 ELSE 0 END) AS overloaded_samples
   810→            FROM recent_logs
   811→            GROUP BY route
   812→            ORDER BY samples DESC, route
   813→            LIMIT 8
   814→            """
   815→        ).fetchall())
   816→        vehicle_rows.extend(dict(row) | {"country": code} for row in conn.execute(
   817→            """
   818→            SELECT route,
   819→                   COUNT(*) AS vehicles,
   820→                   ROUND(AVG(occupancy), 2) AS average_occupancy,
   821→                   SUM(CASE WHEN tier IN ('red', 'blinking_red') THEN 1 ELSE 0 END) AS crowded
   822→            FROM vehicle_states
   823→            GROUP BY route
   824→            ORDER BY vehicles DESC, route
   825→            LIMIT 8
   826→            """
   827→        ).fetchall())
   828→        for row in conn.execute(
   829→            """
   830→            SELECT verification_status, COUNT(*) AS count
   831→            FROM operator_alerts
   832→            GROUP BY verification_status
   833→            ORDER BY count DESC
   834→            """
   835→        ).fetchall():
   836→            alert_totals[row["verification_status"]] = alert_totals.get(row["verification_status"], 0) + int(row["count"])
   837→        recent_chats.extend(dict(row) | {"country": code} for row in conn.execute(
   838→            """
   839→            SELECT route, query, answer, timestamp
   840→            FROM chatbot_queries
   841→            ORDER BY timestamp DESC
   842→            LIMIT 5
   843→            """
   844→        ).fetchall())
   845→    recent_chats = sorted(recent_chats, key=lambda row: row.get("timestamp") or "", reverse=True)[:5]
   846→    alert_rows = [{"verification_status": key, "count": value} for key, value in sorted(alert_totals.items())]
   847→    return {
   848→        "path": str(database_path(countries[0])) if country else str(COUNTRY_DB_DIR),
   849→        "exists": database_path(countries[0]).exists() if country else COUNTRY_DB_DIR.exists(),
   850→        "country": countries[0] if country else None,
   851→        "countries": country_statuses,
   852→        "tables": counts,
   853→        "stats": {
   854→            "telemetry_samples": counts.get("telemetry_logs", 0),
   855→            "active_vehicle_routes": len(vehicle_rows),
   856→            "chat_queries": counts.get("chatbot_queries", 0),
   857→            "open_alerts": sum(row["count"] for row in alert_rows if row["verification_status"] == "open"),
   858→        },
   859→        "route_loads": load_rows[:8],
   860→        "vehicle_routes": vehicle_rows[:8],
   861→        "alert_statuses": alert_rows,
   862→        "recent_chats": recent_chats,
   863→    }

  1064→def load_route_polyline(route: str) -> list[tuple[float, float]]:
  1065→    cache_key = route
  1066→    if cache_key in _ROUTE_POLYLINE_CACHE:
  1067→        return list(_ROUTE_POLYLINE_CACHE[cache_key])
  1068→    row = None
  1069→    for code in COUNTRY_CODES:
  1070→        init_db(code)
  1071→        conn = _get_connection(code)
  1072→        row = conn.execute("SELECT polyline_json FROM routes WHERE route = ? OR tag = ? LIMIT 1", (route, route)).fetchone()
  1073→        if row is not None:
  1074→            break
  1075→    if row is None:
  1076→        return []
  1077→    try:
  1078→        points = json.loads(row["polyline_json"]) if row["polyline_json"] else []
  1079→    except Exception:
  1080→        return []
  1081→    polyline = [(float(lat), float(lon)) for lat, lon in points]
  1082→    _ROUTE_POLYLINE_CACHE[cache_key] = polyline
  1083→    return list(polyline)

  1086→def load_routes(country: str | None = None) -> list[dict[str, Any]]:
  1087→    """Load routes with optional country filter. Uses caching."""
  1088→    global _ROUTE_CACHE
  1089→    if _ROUTE_CACHE is not None:
  1090→        result = _ROUTE_CACHE
  1091→        if country:
  1092→            code = normalize_country(country)
  1093→            result = [r for r in result if (r.get("country") or "") == code]
  1094→        return deepcopy(result)
  1095→    result: list[dict[str, Any]] = []
  1096→    for code in COUNTRY_CODES:
  1097→        init_db(code)
  1098→        conn = _get_connection(code)
  1099→        rows = conn.execute(
  1100→            "SELECT route, name, polyline_json, country, region, tag, route_type, origin_name, destination_name, distance_km, description, minimum_fare, fare_per_km FROM routes"
  1101→        ).fetchall()
  1102→        point_rows = conn.execute(
  1103→            "SELECT route, sequence_order, latitude, longitude, point_type, label FROM route_points ORDER BY route, sequence_order"
  1104→        ).fetchall()
  1105→        points_by_route: dict[str, list[dict[str, Any]]] = {}
  1106→        for point in point_rows:
  1107→            item = dict(point)
  1108→            item["name"] = item.get("label") or item.get("point_type") or f"Point {item.get('sequence_order', '')}".strip()
  1109→            points_by_route.setdefault(point["route"], []).append(item)
  1110→        for row in rows:
  1111→            try:
  1112→                poly = json.loads(row["polyline_json"]) if row["polyline_json"] else []
  1113→            except Exception:
  1114→                poly = []
  1115→            metadata = route_metadata(row["route"])
  1116→            structured_points = points_by_route.get(row["route"], [])
  1117→            stops = structured_points or metadata.get("stops") or _display_stops(poly, row["name"])
  1118→            city = metadata.get("city") or ""
  1119→            endpoints = metadata.get("endpoints") or ([stops[0]["name"], stops[-1]["name"]] if len(stops) >= 2 else [])
  1120→            result.append({
  1121→                "route": row["route"],
  1122→                "name": row["name"],
  1123→                "country": row["country"] or code,
  1124→                "region": row["region"],
  1125→                "tag": row["tag"],
  1126→                "route_type": row["route_type"],
  1127→                "origin_name": row["origin_name"],
  1128→                "destination_name": row["destination_name"],
  1129→                "distance_km": row["distance_km"],
  1130→                "description": row["description"],
  1131→                "minimum_fare": row["minimum_fare"],
  1132→                "fare_per_km": row["fare_per_km"],
  1133→                "polyline": [{"latitude": float(lat), "longitude": float(lon)} for lat, lon in poly],
  1134→                "points": structured_points,
  1135→                "stops": stops,
  1136→                "zone": metadata.get("zone") or city,
  1137→                "type": metadata.get("type") or "PUV",
  1138→                "landmarks": metadata.get("landmarks") or [stop["name"] for stop in stops[:6]],
  1139→                "endpoints": endpoints,
  1140→            })
  1141→    _ROUTE_CACHE = result
  1142→    filtered = result
  1143→    if country:
  1144→        code = normalize_country(country)
  1145→        filtered = [r for r in result if (r.get("country") or "") == code]
  1146→    return deepcopy(filtered)

  1149→def _display_stops(poly: list[list[float]] | list[tuple[float, float]], name: str) -> list[dict[str, Any]]:
  1150→    if not poly:
  1151→        return []
  1152→    if len(poly) <= 8:
  1153→        indexes = list(range(len(poly)))
  1154→    else:
  1155→        indexes = sorted({0, len(poly) - 1, *[round((len(poly) - 1) * ratio) for ratio in (0.2, 0.35, 0.5, 0.65, 0.8)]})
  1156→    labels = ["Origin", "Checkpoint 1", "Checkpoint 2", "Mid-route", "Checkpoint 3", "Checkpoint 4", "Terminal"]
  1157→    stops = []
  1158→    for display_index, point_index in enumerate(indexes):
  1159→        lat, lon = poly[point_index]
  1160→        label = labels[min(display_index, len(labels) - 1)]
  1161→        stops.append({
  1162→            "stop_id": point_index,
  1163→            "name": f"{name} {label}",
  1164→            "latitude": float(lat),
  1165→            "longitude": float(lon),
  1166→        })
  1167→    return stops
