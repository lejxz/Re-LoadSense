     1→function drawMap(containerId, routeFilter) {
     2→    const el = qs(containerId);
     3→    if (!el) return;
     4→    if (typeof L === 'undefined') return;
     5→
     6→    const isMobile = containerId === 'mobileMap';
     7→    const isOperator = containerId === 'operatorMap' || containerId === 'routePreviewMap';
     8→    const isOverviewMap = containerId === "operatorMap" && !routeFilter;
     9→    const isRouteMap = !isOverviewMap && (Boolean(routeFilter) || containerId === "routePreviewMap" || isMobile);
    10→
    11→    // initialize map if needed
    12→    if (!state.maps[containerId]) {
    13→      const map = L.map(containerId, { zoomControl: false, attributionControl: false }).setView([10.3157, 123.8854], 13);
    14→      L.control.zoom({ position: containerId === "operatorMap" ? "topright" : "bottomright" }).addTo(map);
    15→      // Tile layer definitions – Transport is the default
    16→      const TILE_LAYERS = {
    17→        'Transport': L.tileLayer('https://tile.memomaps.de/tilegen/{z}/{x}/{y}.png', {
    18→          maxZoom: 19,
    19→          attribution: '&copy; <a href="https://memomaps.de/">memomaps.de</a> CC-BY-SA, map data &copy; <a href="https://openstreetmap.org/copyright">OpenStreetMap</a> contributors'
    20→        }),
    21→        'Standard': L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    22→          maxZoom: 19,
    23→          attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
    24→        }),
    25→        'Cycle': L.tileLayer('https://{s}.tile-cyclosm.openstreetmap.fr/cyclosm/{z}/{x}/{y}.png', {
    26→          maxZoom: 20,
    27→          attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
    28→        }),
    29→        'Light': L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', {
    30→          maxZoom: 20,
    31→          attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OSM</a> contributors &copy; <a href="https://carto.com/">CARTO</a>'
    32→        }),
    33→        'Dark': L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
    34→          maxZoom: 20,
    35→          attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OSM</a> contributors &copy; <a href="https://carto.com/">CARTO</a>'
    36→        })
    37→      };
    38→      const tileLayer = TILE_LAYERS['Standard'].addTo(map);
    39→      tileLayer.on('tileerror', function(e) {
    40→        if (e.tile) e.tile.style.display = 'none';
    41→      });
    42→      state.maps[containerId] = map;
    43→      state.mapTileLayers[containerId] = TILE_LAYERS;
    44→      state.mapActiveTile[containerId] = tileLayer;
    45→      state.mapLayers[containerId] = L.layerGroup().addTo(map);
    46→      addMapControls(containerId, map, TILE_LAYERS, 'Standard');
    47→      // clustering layer
    48→      try {
    49→        state.mapClusters[containerId] = L.markerClusterGroup({
    50→          iconCreateFunction(cluster) {
    51→            const markers = cluster.getAllChildMarkers();
    52→            const overloaded = markers.filter(marker => marker.options.loadsenseTier === "blinking_red").length;
    53→            const label = overloaded ? `${markers.length}/${overloaded}` : String(markers.length);
    54→            return L.divIcon({ html: `<div><span>${label}</span></div>`, className: "marker-cluster marker-cluster-small", iconSize: L.point(40, 40) });
    55→          },
    56→        });
    57→        state.mapClusters[containerId].addTo(map);
    58→      } catch (e) {
    59→        state.mapClusters[containerId] = null;
    60→      }
    61→      // Track user interaction to disable auto-fit
    62→      map.on('movestart', function (e) {
    63→        // Only set userInteracted for moves not triggered programmatically
    64→        if (!map._programmaticMove) {
    65→          state.userInteracted = true;
    66→        }
    67→      });
    68→    }
    69→    const map = state.maps[containerId];
    70→    const layerGroup = state.mapLayers[containerId];
    71→    const clusterGroup = state.mapClusters[containerId];
    72→    // clear route polylines and route-point layers (redrawn each call)
    73→    layerGroup.clearLayers();
    74→
    75→    // Initialize incremental vehicle marker tracking
    76→    state.vehicleMarkers = state.vehicleMarkers || {};
    77→    if (!state.vehicleMarkers[containerId]) state.vehicleMarkers[containerId] = {};
    78→    const existingMarkers = state.vehicleMarkers[containerId];

    80→    const routes = isOverviewMap ? [] : routeFilter ? state.routes.filter(route => route.route === routeFilter) : state.routes;
    81→    // draw route polylines

    83→      const structuredPoints = [...(route.points || [])].sort((a, b) => Number(a.sequence_order || 0) - Number(b.sequence_order || 0));
    84→      const polyPoints = structuredPoints.length ? structuredPoints : (route.polyline || []);
    85→      const routeStops = structuredPoints.length ? structuredPoints : (route.stops || []);
    86→      // Support both new structured points and legacy [lat, lon] arrays
    87→      const validPoints = polyPoints.filter(p => p && (p.latitude !== undefined || Array.isArray(p)));

    93→            L.polyline(latlngs, { color: '#ffffff', weight: 12, opacity: 0.95, lineCap: 'round', lineJoin: 'round' }).addTo(layerGroup);

    95→            L.polyline(latlngs, { color: '#58a6ff', weight: 3, opacity: 0.95, lineCap: 'round', lineJoin: 'round' }).addTo(layerGroup);
    96→            
    97→            let stopNumber = 0;
    98→            (routeStops.length ? routeStops : validPoints).forEach((point, idx, allPoints) => {
    99→              const isObj = !Array.isArray(point);
   100→              const ptLat = isObj ? Number(point.latitude) : Number(point[0]);
   101→              const ptLon = isObj ? Number(point.longitude) : Number(point[1]);
   102→              let type = isObj ? (point.point_type || 'checkpoint') : 'checkpoint';
   103→              if (idx === 0) type = 'origin';
   104→              if (idx === allPoints.length - 1) type = 'end';
   105→              // Normalize legacy 'turn' type to 'waypoint'
   106→              if (type === 'turn' || type === 'checkpoint') type = 'waypoint';
   107→
   108→              // Show only rider-meaningful pins on mobile to keep the UI clean.
   109→              if (isMobile) {
   110→                if (type !== 'origin' && type !== 'end' && type !== 'end_of_route'
   111→                    && type !== 'alight_or_board_stop' && type !== 'boarding_stop') {
   112→                  return;
   113→                }
   114→              }
   115→
   116→              const label = isObj && point.label ? point.label : type.replace(/_/g, ' ');
   117→              let iconClass, svgIcon, iconSize, iconAnchor;
   118→
   119→              if (type === 'origin') {
   120→                iconClass = 'route-point-icon route-point-origin';
   121→                svgIcon = `<svg viewBox="0 0 32 32" width="24" height="24"><circle cx="16" cy="16" r="14" fill="#16a34a" stroke="#fff" stroke-width="2"/><text x="16" y="22" text-anchor="middle" fill="#fff" font-size="16" font-weight="bold" font-family="sans-serif">A</text></svg>`;
   122→                iconSize = [24, 24];
   123→                iconAnchor = [12, 12];
   124→              } else if (type === 'end' || type === 'end_of_route') {
   125→                iconClass = 'route-point-icon route-point-end';
   126→                svgIcon = `<svg viewBox="0 0 32 32" width="24" height="24"><circle cx="16" cy="16" r="14" fill="#dc2626" stroke="#fff" stroke-width="2"/><text x="16" y="22" text-anchor="middle" fill="#fff" font-size="16" font-weight="bold" font-family="sans-serif">B</text></svg>`;
   127→                iconSize = [24, 24];
   128→                iconAnchor = [12, 12];
   129→              } else if (type === 'alight_or_board_stop' || type === 'boarding_stop') {
   130→                stopNumber++;
   131→                iconClass = 'route-point-icon route-point-stop';
   132→                svgIcon = `<svg viewBox="0 0 28 28" width="20" height="20"><circle cx="14" cy="14" r="12" fill="#2563eb" stroke="#fff" stroke-width="2"/><text x="14" y="19" text-anchor="middle" fill="#fff" font-size="12" font-weight="bold" font-family="sans-serif">${stopNumber}</text></svg>`;
   133→                iconSize = [20, 20];
   134→                iconAnchor = [10, 10];
   135→              } else if (type === 'waypoint') {
   136→                iconClass = 'route-point-icon route-point-waypoint';
   137→                svgIcon = `<svg viewBox="0 0 20 20" width="14" height="14"><rect x="3" y="3" width="14" height="14" rx="2" transform="rotate(45 10 10)" fill="#f97316" stroke="#fff" stroke-width="1.5"/></svg>`;
   138→                iconSize = [14, 14];
   139→                iconAnchor = [7, 7];
   140→              } else if (type === 'road_segment') {
   141→                iconClass = 'route-point-icon route-point-segment';
   142→                svgIcon = `<svg viewBox="0 0 12 12" width="8" height="8"><circle cx="6" cy="6" r="4" fill="#9ca3af"/></svg>`;
   143→                iconSize = [8, 8];
   144→                iconAnchor = [4, 4];
   145→              } else {
   146→                // Fallback: treat unknown types as waypoints
   147→                iconClass = 'route-point-icon route-point-waypoint';
   148→                svgIcon = `<svg viewBox="0 0 20 20" width="14" height="14"><rect x="3" y="3" width="14" height="14" rx="2" transform="rotate(45 10 10)" fill="#f97316" stroke="#fff" stroke-width="1.5"/></svg>`;
   149→                iconSize = [14, 14];
   150→                iconAnchor = [7, 7];
   151→              }
   152→
   153→              const marker = L.marker([ptLat, ptLon], {
   154→                icon: L.divIcon({
   155→                  className: iconClass,
   156→                  html: svgIcon,
   157→                  iconSize: iconSize,
   158→                  iconAnchor: iconAnchor,
   159→                }),
   160→              });
   161→              marker.bindPopup(`<strong>${escapeHtml(label.charAt(0).toUpperCase() + label.slice(1))}</strong><br/><small>${escapeHtml(type.replace(/_/g, ' '))}</small><br/><small>LAT: ${ptLat.toFixed(5)}, LONG: ${ptLon.toFixed(5)}</small>`)
   162→                    .bindTooltip(escapeHtml(label), { direction: 'top', offset: [0, -5] });
   163→              marker.addTo(layerGroup);
   164→            });
   165→          } else if (!isOverviewMap) {
   166→            L.polyline(latlngs, { color: '#9aa0a6', weight: 2, opacity: routeFilter ? 0.25 : 0.16, lineCap: 'round', lineJoin: 'round' }).addTo(layerGroup);
   167→          }
   168→      }
   169→    }
   170→    // Incremental vehicle marker updates
   171→    const vehicles = state.vehicles.filter(vehicle => !routeFilter || vehicle.route === routeFilter).filter(vehicle => Number(vehicle.latitude) && Number(vehicle.longitude));
   172→    const currentVehicleIds = new Set();
   173→    for (const vehicle of vehicles) {
   174→      const lat = Number(vehicle.latitude);
   175→      const lon = Number(vehicle.longitude);
   176→      const tier = tierClass(vehicle.tier);
   177→      const vid = vehicle.vehicle_id;
   178→      currentVehicleIds.add(vid);
   179→
   180→      if (existingMarkers[vid]) {
   181→        // Update existing marker position and icon
   182→        const existing = existingMarkers[vid];
   183→        existing.marker.setLatLng([lat, lon]);
   184→        // Update icon if tier changed
   185→        if (existing.tier !== vehicle.tier) {
   186→          const newIcon = L.divIcon({
   187→            className: 'vehicle-icon-wrapper',
   188→            html: `<span class="vehicle-div-icon ${tier}" aria-hidden="true"><svg viewBox="0 0 24 24"><path d="M6 3h12a3 3 0 0 1 3 3v9a2 2 0 0 1-2 2v2a1 1 0 0 1-2 0v-2H7v2a1 1 0 0 1-2 0v-2a2 2 0 0 1-2-2V6a3 3 0 0 1 3-3Zm0 3v5h12V6H6Zm2 8a1.5 1.5 0 1 0 0 .01V14Zm8 0a1.5 1.5 0 1 0 0 .01V14Z"/></svg></span>`,
   189→            iconSize: [28, 28],
   190→            iconAnchor: [14, 14],
   191→            popupAnchor: [0, -14]
   192→          });
   193→          existing.marker.setIcon(newIcon);
   194→          existing.tier = vehicle.tier;
   195→        }
   196→        // Update popup content
   197→        const popup = `<div class="map-popup-actions"><strong>${escapeHtml(vehicle.vehicle_id)}</strong><span>Route ${escapeHtml(vehicle.route)}</span><span>Load ${escapeHtml(String(vehicle.occupancy))}/${escapeHtml(String(vehicle.capacity))} · ETA ${escapeHtml(String(vehicle.eta_minutes))} min</span><div><button onclick="showVehicleDetailsModal('${escapeHtml(vehicle.vehicle_id)}')">Details</button><button onclick="reportVehicleIncidentModal('${escapeHtml(vehicle.vehicle_id)}','${escapeHtml(vehicle.route)}')">Report</button></div></div>`;
   198→        existing.marker.setPopupContent(popup);
   199→      } else {
   200→        // Create new marker
   201→        const icon = L.divIcon({
   202→          className: 'vehicle-icon-wrapper',
   203→          html: `<span class="vehicle-div-icon ${tier}" aria-hidden="true"><svg viewBox="0 0 24 24"><path d="M6 3h12a3 3 0 0 1 3 3v9a2 2 0 0 1-2 2v2a1 1 0 0 1-2 0v-2H7v2a1 1 0 0 1-2 0v-2a2 2 0 0 1-2-2V6a3 3 0 0 1 3-3Zm0 3v5h12V6H6Zm2 8a1.5 1.5 0 1 0 0 .01V14Zm8 0a1.5 1.5 0 1 0 0 .01V14Z"/></svg></span>`,
   204→          iconSize: [28, 28],
   205→          iconAnchor: [14, 14],
   206→          popupAnchor: [0, -14]
   207→        });
   208→        const marker = L.marker([lat, lon], { icon, loadsenseTier: vehicle.tier });
   209→        const popup = `<div class="map-popup-actions"><strong>${escapeHtml(vehicle.vehicle_id)}</strong><span>Route ${escapeHtml(vehicle.route)}</span><span>Load ${escapeHtml(String(vehicle.occupancy))}/${escapeHtml(String(vehicle.capacity))} · ETA ${escapeHtml(String(vehicle.eta_minutes))} min</span><div><button onclick="showVehicleDetailsModal('${escapeHtml(vehicle.vehicle_id)}')">Details</button><button onclick="reportVehicleIncidentModal('${escapeHtml(vehicle.vehicle_id)}','${escapeHtml(vehicle.route)}')">Report</button></div></div>`;
   210→        marker.bindPopup(popup);
   211→        marker.bindTooltip(`${escapeHtml(vehicle.vehicle_id)} (${escapeHtml(vehicle.route)})` , { permanent: false });
   212→        if (clusterGroup) {
   213→          clusterGroup.addLayer(marker);
   214→        } else {
   215→          marker.addTo(layerGroup);
   216→        }
   217→        existingMarkers[vid] = { marker, tier: vehicle.tier };
   218→      }
   219→    }
   220→    // Remove markers for vehicles that no longer exist
   221→    for (const vid of Object.keys(existingMarkers)) {
   222→      if (!currentVehicleIds.has(vid)) {
   223→        const entry = existingMarkers[vid];
   224→        if (clusterGroup) {
   225→          clusterGroup.removeLayer(entry.marker);
   226→        } else {
   227→          layerGroup.removeLayer(entry.marker);
   228→        }
   229→        delete existingMarkers[vid];
   230→      }
   231→    }
   232→    drawDestinationLayer(containerId, layerGroup);
   233→    // Only fitBounds when user has not interacted with the map
   234→    if (!state.userInteracted && vehicles.length) {
   235→      const allPoints = vehicles.map(v => [Number(v.latitude), Number(v.longitude)]);
   236→      map._programmaticMove = true;
   237→      try { map.fitBounds(L.latLngBounds(allPoints), { maxZoom: 15, padding: [28, 28] }); } catch (e) {}
   238→      map._programmaticMove = false;
   239→    }
   240→  }

   259→  function fitRoute(containerId, routeId) {
   260→    const map = state.maps[containerId];
   261→    const route = state.routes.find(item => item.route === routeId);
   262→    const points = (route?.polyline || []).filter(isMapCoordinate).map(point => [Number(point.latitude), Number(point.longitude)]);
   263→    if (map && points.length && typeof L !== "undefined") {
   264→      try { map.fitBounds(L.latLngBounds(points), { maxZoom: 15, padding: [28, 28] }); } catch (e) {}
   265→    }
   266→  }

   276→  function previewRoute(routeId, containerId = "operatorMap") {
   277→    const route = state.routes.find(item => item.route === routeId);
   278→    if (!route) return;
   279→    state.selectedRoute = routeId;
   280→    const modal = qs("routePreviewModal");
   281→    if (modal && containerId === "routePreviewMap") {
   282→      modal.classList.remove("hidden");
   283→      qs("routePreviewTitle").textContent = routeDisplayTitle(route);
   284→      const summary = routeSummary(route);
   285→      qs("routePreviewMeta").textContent = `${cityName(route)} - ${summary.stopCount} stops - ${summary.vehicleCount} live PUVs`;
   286→    }
   287→    drawMap(containerId, routeId);
   288→    setTimeout(() => {
   289→      fitRoute(containerId, routeId);
   290→      if (containerId === "routePreviewMap" && state.maps[containerId]) {
   291→        state.maps[containerId].invalidateSize();
   292→      }
   293→    }, 120);
   294→  }

   296→  function zoomVehicle(vehicleId) {
   297→    const vehicle = state.vehicles.find(item => item.vehicle_id === vehicleId);
   298→    if (!vehicle || !isMapCoordinate(vehicle)) return;
   299→    state.selectedVehicleId = vehicleId;
   300→    state.selectedRoute = vehicle.route;
   301→    
   302→    if (state.maps.mobileMap) {

   306→      if (map) {
   307→        setTimeout(() => map.setView([Number(vehicle.latitude), Number(vehicle.longitude)], 17), 100);
   308→      }

   310→      if (typeof activateOperatorTab === "function") activateOperatorTab("opsOverview");

   312→      if (map) {
   313→        setTimeout(() => map.setView([Number(vehicle.latitude), Number(vehicle.longitude)], 17), 100);
   314→      }
   315→    }

   318→  function addMapControls(containerId, map, tileLayers, defaultName) {
   319→    if (typeof L === "undefined") return;
   320→    const isOperator = containerId === "operatorMap";
   321→    const controlPos = isOperator ? "topright" : "bottomright";
   322→    const control = L.control({ position: controlPos });
   323→    control.onAdd = function () {
   324→      const wrap = L.DomUtil.create("div", "map-controls-bar");
   325→      L.DomEvent.disableClickPropagation(wrap);
   326→      L.DomEvent.disableScrollPropagation(wrap);
   327→
   328→      // --- Layer Settings Bottom Sheet ---
   329→      const mapContainer = map.getContainer();
   330→      const sheetWrap = L.DomUtil.create("div", "map-settings-sheet-wrap", mapContainer);
   331→      const backdrop = L.DomUtil.create("div", "map-settings-backdrop", sheetWrap);
   332→      const sheet = L.DomUtil.create("div", "map-settings-sheet", sheetWrap);
   333→
   334→      const sheetHeader = L.DomUtil.create("div", "map-settings-header", sheet);
   335→      sheetHeader.innerHTML = `<h3>Map Details</h3><button type="button" class="close-sheet" aria-label="Close Map Details"><svg width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><line x1="18" y1="6" x2="6" y2="18"></line><line x1="6" y1="6" x2="18" y2="18"></line></svg></button>`;
   336→      
   337→      const grid = L.DomUtil.create("div", "map-settings-grid", sheet);
   338→      const layerOrder = ['Standard', 'Light', 'Dark', 'Transport', 'Cycle'];
   339→      
   340→      layerOrder.forEach(name => {
   341→        const item = L.DomUtil.create("button", "map-type-card", grid);
   342→        item.type = "button";
   343→        item.innerHTML = `<div class="map-type-preview map-type-${name.toLowerCase()}"></div><div class="map-type-name">${name}</div>`;
   344→        if (name === defaultName) item.classList.add("active");
   345→        
   346→        L.DomEvent.on(item, "click", (e) => {
   347→          e.stopPropagation();
   348→          Array.from(grid.children).forEach(child => child.classList.remove("active"));
   349→          item.classList.add("active");
   350→          closeSheet();
   351→          
   352→          const current = state.mapActiveTile[containerId];
   353→          if (current) map.removeLayer(current);
   354→          const next = tileLayers[name];
   355→          next.addTo(map);
   356→          const featureLayer = state.mapLayers[containerId];
   357→          const clusterLayer = state.mapClusters[containerId];
   358→          if (featureLayer && featureLayer.bringToFront) featureLayer.bringToFront();
   359→          if (clusterLayer && clusterLayer.bringToFront) clusterLayer.bringToFront();
   360→          state.mapActiveTile[containerId] = next;
   361→        });
   362→      });
   363→
   364→      const openSheet = () => {
   365→        backdrop.classList.add("open");
   366→        sheet.classList.add("open");
   367→      };
   368→      const closeSheet = () => {
   369→        backdrop.classList.remove("open");
   370→        sheet.classList.remove("open");
   371→      };
   372→      L.DomEvent.on(backdrop, "click", closeSheet);
   373→      L.DomEvent.on(sheetHeader.querySelector(".close-sheet"), "click", closeSheet);
   374→
   375→
   376→
   377→      // --- Choose on Map (Pin Destination) Button ---
   378→      if (containerId === "mobileMap") {
   379→        const pinBtn = L.DomUtil.create("button", "map-fab pin-dest-btn", wrap);
   380→        pinBtn.type = "button";
   381→        pinBtn.title = "Choose destination on map";
   382→        pinBtn.setAttribute("aria-label", "Choose destination on map");
   383→        pinBtn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.8" stroke-linecap="round" stroke-linejoin="round" style="transform: scale(0.9)"><path d="M22 2 11 13"></path><path d="m22 2-7 20-4-9-9-4 20-7z"></path></svg>`;
   384→        
   385→        L.DomEvent.on(pinBtn, "click", (e) => {
   386→          e.stopPropagation();
   387→          if (typeof window.togglePinMode === 'function') {
   388→            window.togglePinMode(containerId, map);

   464→  function startLocationWatch(containerId, map) {
   465→    // Avoid duplicate watches for the same map
   466→    if (state.geoWatches && state.geoWatches[containerId]) return;
   467→    if (!state.geoWatches) state.geoWatches = {};
   468→    let locationMarker = null;
   469→    let accuracyCircle = null;
   470→    state.geoWatches[containerId] = navigator.geolocation.watchPosition(
   471→      (pos) => {
   472→        const lat = pos.coords.latitude;
   473→        const lon = pos.coords.longitude;
   474→        const acc = pos.coords.accuracy;
   475→        // Update shared state
   476→        state.lastPosition = { latitude: lat, longitude: lon };
   477→        // Draw / update marker and accuracy circle
   478→        if (!locationMarker) {
   479→          accuracyCircle = L.circle([lat, lon], {
   480→            radius: acc,
   481→            color: '#1a73e8',
   482→            fillColor: '#1a73e8',
   483→            fillOpacity: 0.08,
   484→            weight: 1.5,
   485→            opacity: 0.4,
   486→          }).addTo(map);
   487→          locationMarker = L.marker([lat, lon], {
   488→            icon: L.divIcon({
   489→              className: 'my-location-icon',
   490→              html: '<span></span>',
   491→              iconSize: [18, 18],
   492→              iconAnchor: [9, 9],
   493→            }),
   494→            zIndexOffset: 1000,
   495→          }).bindTooltip('My location', { direction: 'top', offset: [0, -6] }).addTo(map);
   496→        } else {
   497→          locationMarker.setLatLng([lat, lon]);
   498→          accuracyCircle.setLatLng([lat, lon]).setRadius(acc);
   499→        }
   500→      },
   501→      null,
   502→      { enableHighAccuracy: true, maximumAge: 10000, timeout: 15000 }
   503→    );
   504→  }
