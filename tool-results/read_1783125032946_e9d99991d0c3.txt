   593‚Üí   593‚Üídef validate_polyline(polyline: list[list[float]]) -> None:
   594‚Üí   594‚Üí    if len(polyline) < 2:
   595‚Üí   595‚Üí        raise ValueError("polyline must contain at least two points")
   596‚Üí   596‚Üí    for index, point in enumerate(polyline):
   597‚Üí   597‚Üí        if len(point) != 2:
   598‚Üí   598‚Üí            raise ValueError(f"point {index} must be [latitude, longitude]")
   599‚Üí   599‚Üí        lat, lon = point
   600‚Üí   600‚Üí        if not all(isinstance(value, (int, float)) and math.isfinite(value) for value in [lat, lon]):
   601‚Üí   601‚Üí            raise ValueError(f"point {index} contains non-numeric coordinates")
   602‚Üí   602‚Üí        if not -90 <= float(lat) <= 90 or not -180 <= float(lon) <= 180:
   603‚Üí   603‚Üí            raise ValueError(f"point {index} is outside latitude/longitude bounds")
   604‚Üí   604‚Üí

   606‚Üí   606‚Üídef validate_imported_routes(routes: list[dict[str, Any]], replace: bool = False) -> list[str]:
   607‚Üí   607‚Üí    errors: list[str] = []
   608‚Üí   608‚Üí    seen_routes: set[str] = set()
   609‚Üí   609‚Üí    seen_names: set[str] = set()
   610‚Üí   610‚Üí    for index, route in enumerate(routes):
   611‚Üí   611‚Üí        route_id = str(route.get("route", "")).strip()
   612‚Üí   612‚Üí        name = str(route.get("name", "")).strip()
   613‚Üí   613‚Üí        if not route_id or not name:
   614‚Üí   614‚Üí            errors.append(f"route {index + 1}: route and name are required")
   615‚Üí   615‚Üí            continue
   616‚Üí   616‚Üí        if route_id.lower() in seen_routes:
   617‚Üí   617‚Üí            errors.append(f"route {route_id}: duplicate route id inside import")
   618‚Üí   618‚Üí        if name.lower() in seen_names:
   619‚Üí   619‚Üí            errors.append(f"route {route_id}: duplicate route name inside import")
   620‚Üí   620‚Üí        seen_routes.add(route_id.lower())
   621‚Üí   621‚Üí        seen_names.add(name.lower())
   622‚Üí   622‚Üí        try:
   623‚Üí   623‚Üí            validate_polyline(route.get("polyline", []))
   624‚Üí   624‚Üí        except ValueError as exc:
   625‚Üí   625‚Üí            errors.append(f"route {route_id}: {exc}")
   626‚Üí   626‚Üí        if sqlite_store.route_exists(route_id) and not replace:

   683‚Üí   683‚Üídef parse_gtfs_routes(content: bytes) -> list[dict[str, Any]]:

   685‚Üí   685‚Üí        names = set(archive.namelist())
   686‚Üí   686‚Üí        if "shapes.txt" not in names:
   687‚Üí   687‚Üí            raise ValueError("GTFS zip must include shapes.txt")

   690‚Üí   690‚Üí            with archive.open("routes.txt") as handle:

   697‚Üí   697‚Üí                for row in csv.DictReader(io.TextIOWrapper(handle, encoding="utf-8-sig")):
   698‚Üí   698‚Üí                    if row.get("shape_id") and row.get("route_id"):
   699‚Üí   699‚Üí                        shape_to_route.setdefault(row["shape_id"], row["route_id"])
   700‚Üí   700‚Üí        shapes: dict[str, list[tuple[int, float, float]]] = {}
   701‚Üí   701‚Üí        with archive.open("shapes.txt") as handle:
   702‚Üí   702‚Üí            for row in csv.DictReader(io.TextIOWrapper(handle, encoding="utf-8-sig")):
   703‚Üí   703‚Üí                shape_id = row["shape_id"]

   705‚Üí   705‚Üí                shapes.setdefault(shape_id, []).append((sequence, float(row["shape_pt_lat"]), float(row["shape_pt_lon"])))

   707‚Üí   707‚Üí    for shape_id, points in shapes.items():
   708‚Üí   708‚Üí        route_id = shape_to_route.get(shape_id, shape_id)

   710‚Üí   710‚Üí        routes.append({
   711‚Üí   711‚Üí            "route": route_id,
   712‚Üí   712‚Üí            "name": route_names.get(route_id, route_id),
   713‚Üí   713‚Üí            "polyline": [[lat, lon] for _, lat, lon in points],
   714‚Üí   714‚Üí        })
   715‚Üí   715‚Üí    return routes

   718‚Üí   718‚Üídef simplify_polyline(points: list[list[float]], tolerance: float) -> list[list[float]]:
   719‚Üí   719‚Üí    if len(points) <= 2:
   720‚Üí   720‚Üí        return points
   721‚Üí   721‚Üí
   722‚Üí   722‚Üí    def perpendicular_distance(point: list[float], start: list[float], end: list[float]) -> float:
   723‚Üí   723‚Üí        if start == end:
   724‚Üí   724‚Üí            return math.hypot(point[0] - start[0], point[1] - start[1])
   725‚Üí   725‚Üí        numerator = abs((end[1] - start[1]) * point[0] - (end[0] - start[0]) * point[1] + end[0] * start[1] - end[1] * start[0])
   726‚Üí   726‚Üí        denominator = math.hypot(end[1] - start[1], end[0] - start[0])
   727‚Üí   727‚Üí        return numerator / denominator
   728‚Üí   728‚Üí
   729‚Üí   729‚Üí    max_distance = 0.0
   730‚Üí   730‚Üí    max_index = 0
   731‚Üí   731‚Üí    for index in range(1, len(points) - 1):
   732‚Üí   732‚Üí        distance = perpendicular_distance(points[index], points[0], points[-1])
   733‚Üí   733‚Üí        if distance > max_distance:
   734‚Üí   734‚Üí            max_index = index
   735‚Üí   735‚Üí            max_distance = distance
   736‚Üí   736‚Üí    if max_distance > tolerance:
   737‚Üí   737‚Üí        left = simplify_polyline(points[: max_index + 1], tolerance)
   738‚Üí   738‚Üí        right = simplify_polyline(points[max_index:], tolerance)
   739‚Üí   739‚Üí        return left[:-1] + right
   740‚Üí   740‚Üí    return [points[0], points[-1]]
